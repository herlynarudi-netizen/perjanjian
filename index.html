// --- LOGIKA EXPORT PDF (SOLUSI FINAL UNTUK HP) --- document.getElementById('btnSimpanPdf').addEventListener('click', () => { const dokumenPdf = document.getElementById('dokumenPdf'); const loadingMsg = document.getElementById('loadingMsg'); let namaPengemudi = document.getElementById('namaMitra').value.trim(); if(!namaPengemudi) namaPengemudi = "Mang_Sopir"; const fileName = `Surat_Perjanjian_${namaPengemudi}.pdf`; loadingMsg.style.display = 'block'; // 1. Gulir ke atas secara instan window.scrollTo(0, 0); // 2. Beri jeda 300 milidetik agar browser HP selesai menggambar ulang (mencegah bug layar terpotong) setTimeout(() => { html2canvas(dokumenPdf, { scale: 2, useCORS: true, // Trik jitu untuk bypass bug viewport mobile: scrollY: -window.scrollY, // Paksa html2canvas membaca tinggi full dari elemen, bukan layar HP height: dokumenPdf.scrollHeight, windowHeight: dokumenPdf.scrollHeight }).then(canvasImage => { const imgData = canvasImage.toDataURL('image/png'); const { jsPDF } = window.jspdf; const pdf = new jsPDF('p', 'mm', 'a4'); const a4Width = pdf.internal.pageSize.getWidth(); const a4Height = pdf.internal.pageSize.getHeight(); // Hitung rasio agar fit ke kertas A4 const ratio = Math.min(a4Width / canvasImage.width, a4Height / canvasImage.height); const finalWidth = canvasImage.width * ratio; const finalHeight = canvasImage.height * ratio; const marginX = (a4Width - finalWidth) / 2; const marginY = 10; pdf.addImage(imgData, 'PNG', marginX, marginY, finalWidth, finalHeight); pdf.save(fileName); loadingMsg.style.display = 'none'; }).catch(err => { console.error("Gagal membuat PDF: ", err); alert("Terjadi kesalahan saat memproses PDF."); loadingMsg.style.display = 'none'; }); }, 300); // Jeda 300ms });
